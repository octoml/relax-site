49  inline std::string GetUniqueName(std::string prefix) {
50  std::replace(prefix.begin(), prefix.end(), '.', '_');
51  std::string unique_prefix = prefix;
52  auto it = alloc_map_.find(prefix);
53  if (it != alloc_map_.end()) {
54  while (alloc_map_.count(unique_prefix = prefix + std::to_string(++it->second)) > 0) {
55  }
56  }
57  alloc_map_[unique_prefix] = 0;
58  return unique_prefix;
59  }

61  NameTable() = default;

63  template <typename Iter, typename Lambda>
64  explicit NameTable(Iter begin, Iter end, Lambda f) {
65  // static_assert is more reader-friendly than SFINAE when template specialization is not needed.
66  static_assert(std::is_convertible<decltype(f(*begin)), std::string>::value,
67  "Lambda f must has a signature of [?](*it) -> string {}");
68  for (auto it = begin; it != end; ++it) {
69  const std::string& name = f(*it);
70  const size_t idx_last_first_num = std::distance(
71  std::find_if(name.rbegin(), name.rend(), [](char c) { return !std::isdigit(c); }),
72  name.rend());
73  // name = {O = others}{D = consecutive digits}
74  // let O -> prefix;
75  std::string prefix = name.substr(0, idx_last_first_num);
76  ICHECK(prefix.size() > 0 && std::isalpha(prefix[0])) << "Invalid variable name: " << name;
77  if (0 == alloc_map_.count(prefix)) alloc_map_[prefix] = 0;
78  if (idx_last_first_num < name.size()) { // has some digits.
79  // let D's nearest natural number -> idx;
80  // note: stoul("000123") = 123;
81  alloc_map_[prefix] =
82  std::max(alloc_map_[prefix], std::stoi(name.substr(idx_last_first_num)));
83  }
84  }
85  }
