109 template <typename R, typename... Args>
110 class ExprFunctor<R(const Expr& n, Args...)> {
111  private:
112  using TSelf = ExprFunctor<R(const Expr& n, Args...)>;
113  using FType = tvm::NodeFunctor<R(const ObjectRef& n, TSelf* self, Args...)>;
114 
115  public:
117  using result_type = R;
119  virtual ~ExprFunctor() {}
126  R operator()(const Expr& n, Args... args) { return VisitExpr(n, std::forward<Args>(args)...); }
133  virtual R VisitExpr(const Expr& n, Args... args) {
134  ICHECK(n.defined()) << "Found null pointer node while traversing AST. The previous pass may "
135  "have generated invalid data.";
136  static FType vtable = InitVTable();
137  return vtable(n, this, std::forward<Args>(args)...);
138  }
139  // Functions that can be overriden by subclass
140  // NOTE: cross dialect calls are invoked through global var
141  // We do not expect inline PrimFunc to appear in relax IR.
142  virtual R VisitExpr_(const ConstantNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
143  virtual R VisitExpr_(const TupleNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
144  virtual R VisitExpr_(const VarNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
145  virtual R VisitExpr_(const DataflowVarNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
146  virtual R VisitExpr_(const ShapeExprNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
147  virtual R VisitExpr_(const ExternFuncNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
148  virtual R VisitExpr_(const GlobalVarNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
149  virtual R VisitExpr_(const FunctionNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
150  virtual R VisitExpr_(const CallNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
151  virtual R VisitExpr_(const SeqExprNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
152  virtual R VisitExpr_(const IfNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
153  virtual R VisitExpr_(const OpNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
154  virtual R VisitExpr_(const TupleGetItemNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
155  virtual R VisitExpr_(const PrimValueNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
156  virtual R VisitExpr_(const StringImmNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
157  virtual R VisitExpr_(const DataTypeImmNode* op, Args... args) EXPR_FUNCTOR_DEFAULT;
158  virtual R VisitExprDefault_(const Object* op, Args...) {
159  LOG(FATAL) << "Do not have a default for " << op->GetTypeKey();
160  throw;
161  }
162 
163  private:
164  // initialize the vtable.
165  static FType InitVTable() {
166  FType vtable;
167  // Set dispatch
168  RELAX_EXPR_FUNCTOR_DISPATCH(ConstantNode);
169  RELAX_EXPR_FUNCTOR_DISPATCH(TupleNode);
170  RELAX_EXPR_FUNCTOR_DISPATCH(VarNode);
171  RELAX_EXPR_FUNCTOR_DISPATCH(DataflowVarNode);
172  RELAX_EXPR_FUNCTOR_DISPATCH(ShapeExprNode);
173  RELAX_EXPR_FUNCTOR_DISPATCH(ExternFuncNode);
174  RELAX_EXPR_FUNCTOR_DISPATCH(GlobalVarNode);
175  RELAX_EXPR_FUNCTOR_DISPATCH(FunctionNode);
176  RELAX_EXPR_FUNCTOR_DISPATCH(CallNode);
177  RELAX_EXPR_FUNCTOR_DISPATCH(SeqExprNode);
178  RELAX_EXPR_FUNCTOR_DISPATCH(IfNode);
179  RELAX_EXPR_FUNCTOR_DISPATCH(OpNode);
180  RELAX_EXPR_FUNCTOR_DISPATCH(TupleGetItemNode);
181  RELAX_EXPR_FUNCTOR_DISPATCH(PrimValueNode);
182  RELAX_EXPR_FUNCTOR_DISPATCH(StringImmNode);
183  RELAX_EXPR_FUNCTOR_DISPATCH(DataTypeImmNode);
184  return vtable;
185  }
