96 class CallNode : public ExprNode {
97  public:
104  Expr op;
105 
107  tvm::Array<Expr> args;
108 
110  Attrs attrs;
111 
118  Array<StructInfo> sinfo_args;
119 
120  void VisitAttrs(tvm::AttrVisitor* v) {
121  v->Visit("op", &op);
122  v->Visit("args", &args);
123  v->Visit("attrs", &attrs);
124  v->Visit("sinfo_args", &sinfo_args);
125  v->Visit("struct_info_", &struct_info_);
126  v->Visit("_checked_type_", &checked_type_);
127  v->Visit("span", &span);
128  }
129 
130  bool SEqualReduce(const CallNode* other, SEqualReducer equal) const {
131  // skip sinfo_args check for primitive ops.
132  equal->MarkGraphNode();
133  return equal(op, other->op) && equal(args, other->args) && equal(attrs, other->attrs) &&
134  (IsPrimitiveOp(op) || equal(sinfo_args, other->sinfo_args)) &&
135  equal(struct_info_, other->struct_info_);
136  }
137 
138  void SHashReduce(SHashReducer hash_reduce) const {
139  hash_reduce->MarkGraphNode();
140  hash_reduce(op);
141  hash_reduce(args);
142  hash_reduce(attrs);
143  if (!IsPrimitiveOp(op)) {
144  hash_reduce(sinfo_args);
145  }
146  hash_reduce(struct_info_);
147  }
148 
149  static constexpr const char* _type_key = "relax.expr.Call";
150  TVM_DECLARE_FINAL_OBJECT_INFO(CallNode, ExprNode);
151 };
