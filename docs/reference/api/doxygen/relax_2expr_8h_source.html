97 class CallNode : public ExprNode {
98  public:
105  Expr op;
106 
108  tvm::Array<Expr> args;
109 
111  Attrs attrs;
112 
131  tvm::Array<Type> type_args;
132 
133  void VisitAttrs(tvm::AttrVisitor* v) {
134  v->Visit("op", &op);
135  v->Visit("args", &args);
136  v->Visit("attrs", &attrs);
137  v->Visit("type_args", &type_args);
138  v->Visit("struct_info_", &struct_info_);
139  v->Visit("_checked_type_", &checked_type_);
140  v->Visit("span", &span);
141  }
142 
143  bool SEqualReduce(const CallNode* other, SEqualReducer equal) const {
144  // skip type_args check for primitive ops.
145  equal->MarkGraphNode();
146  return equal(op, other->op) && equal(args, other->args) && equal(attrs, other->attrs) &&
147  (IsPrimitiveOp(op) || equal(type_args, other->type_args)) &&
148  equal(struct_info_, other->struct_info_);
149  }
150 
151  void SHashReduce(SHashReducer hash_reduce) const {
152  hash_reduce->MarkGraphNode();
153  hash_reduce(op);
154  hash_reduce(args);
155  hash_reduce(attrs);
156  if (!IsPrimitiveOp(op)) {
157  hash_reduce(type_args);
158  }
159  hash_reduce(struct_info_);
160  }
161 
162  static constexpr const char* _type_key = "relax.expr.Call";
163  TVM_DECLARE_FINAL_OBJECT_INFO(CallNode, ExprNode);
164 };
